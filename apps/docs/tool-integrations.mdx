# Tool Integration Patterns

HyperDev is designed to integrate seamlessly into your existing development environment rather than replace your current toolchain. This guide covers comprehensive integration patterns that make HyperDev feel like a native part of your workflow.

### Vim/Neovim Integration

#### Vim Plugin Configuration

```lua
-- ~/.config/nvim/lua/hyperdev.lua
local M = {}

-- HyperDev command integration
function M.generate_component()
  local component_name = vim.fn.input('Component name: ')
  if component_name ~= '' then
    vim.cmd('!hyperdev gen component ' .. component_name)
    vim.cmd('edit')
  end
end

function M.validate_templates()
  vim.cmd('!hyperdev validate --all')
end

function M.start_dev_mode()
  vim.cmd('terminal hyperdev dev --watch')
end

-- Key mappings
vim.keymap.set('n', '<leader>hc', M.generate_component, { desc = 'HyperDev: Generate Component' })
vim.keymap.set('n', '<leader>hv', M.validate_templates, { desc = 'HyperDev: Validate Templates' })
vim.keymap.set('n', '<leader>hd', M.start_dev_mode, { desc = 'HyperDev: Dev Mode' })

return M
```

#### Autocomplete Integration

```lua
-- ~/.config/nvim/lua/hyperdev-cmp.lua
local cmp = require('cmp')

-- HyperDev template autocomplete source
local source = {}

function source:complete(request, callback)
  local templates = vim.fn.systemlist('hyperdev list --format json')
  local items = {}
  
  for _, template in ipairs(templates) do
    table.insert(items, {
      label = template.name,
      kind = cmp.lsp.CompletionItemKind.Snippet,
      detail = template.description,
      insertText = 'hyperdev gen ' .. template.name .. ' ${1:name}'
    })
  end
  
  callback({ items = items })
end

cmp.register_source('hyperdev', source)
```

## CI/CD Integration

### GitHub Actions Integration

```yaml
# .github/workflows/hyperdev-validation.yml
name: HyperDev Validation

on:
  push:
    branches: [ main, develop ]
    paths:
      - '.hyperdev/**'
      - '**/templates/**'
  pull_request:
    branches: [ main ]

jobs:
  validate-templates:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          
      - name: Install HyperDev
        run: |
          npm install -g @hyperdev/cli
          hyperdev --version
          
      - name: Validate Templates
        run: |
          hyperdev validate --all --format json > template-validation.json
          
      - name: Test Template Generation
        run: |
          # Test critical templates
          hyperdev gen component TestComponent --dry-run
          hyperdev gen api-route test-route --dry-run
          hyperdev gen page test-page --dry-run
          
      - name: Security Scan Templates
        run: |
          hyperdev security-scan --templates --report security-report.json
          
      - name: Upload Validation Results
        uses: actions/upload-artifact@v4
        with:
          name: hyperdev-validation-results
          path: |
            template-validation.json
            security-report.json

  integration-test:
    needs: validate-templates
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [18, 20]
        template: [component, api-route, page]
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}
          
      - name: Test Template Generation
        run: |
          hyperdev gen ${{ matrix.template }} test-${{ matrix.template }} --output temp/
          
      - name: Validate Generated Code
        run: |
          cd temp/
          npm install
          npm run lint
          npm run type-check
          npm run test
```

### Quality Gates in CI/CD

```yaml
# .github/workflows/quality-gates.yml
name: HyperDev Quality Gates

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  quality-gates:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Run HyperDev Quality Gates
        run: |
          # TypeScript validation
          hyperdev quality-gate typescript --strict
          
          # ESLint validation
          hyperdev quality-gate eslint --max-warnings 0
          
          # Security scanning
          hyperdev quality-gate security --fail-on-high
          
          # Performance validation
          hyperdev quality-gate performance --threshold 90
          
          # Architecture compliance
          hyperdev quality-gate architecture --rules .hyperdev/architecture.yml
          
          # Generate quality report
          hyperdev quality-gate report --format json --output quality-report.json
          
      - name: Upload Quality Report
        uses: actions/upload-artifact@v4
        with:
          name: quality-report
          path: quality-report.json
          
      - name: Comment PR with Results
        if: github.event_name == 'pull_request'
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            const report = JSON.parse(fs.readFileSync('quality-report.json'));
            
            const comment = `## HyperDev Quality Gates Results
            
            | Gate | Status | Score |
            |------|--------|--------|
            | TypeScript | ${report.typescript.passed ? '✅' : '❌'} | ${report.typescript.score} |
            | ESLint | ${report.eslint.passed ? '✅' : '❌'} | ${report.eslint.score} |
            | Security | ${report.security.passed ? '✅' : '❌'} | ${report.security.score} |
            | Performance | ${report.performance.passed ? '✅' : '❌'} | ${report.performance.score} |
            | Architecture | ${report.architecture.passed ? '✅' : '❌'} | ${report.architecture.score} |
            
            **Overall Score: ${report.overall.score}/100**
            `;
            
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: comment
            });
```

## Development Workflow Integration

### Git Hooks Integration

#### Pre-commit Hooks

```bash
#!/bin/sh
# .git/hooks/pre-commit

echo "Running HyperDev pre-commit validations..."

# Validate templates before commit
if ! hyperdev validate --changed-only --quiet; then
    echo "❌ Template validation failed. Please fix templates before committing."
    exit 1
fi

# Check for security issues in templates
if ! hyperdev security-scan --changed-only --fail-on-medium; then
    echo "❌ Security issues found in templates. Please address before committing."
    exit 1
fi

# Validate generated code is up to date
if ! hyperdev verify-generated --fail-on-outdated; then
    echo "❌ Generated code is outdated. Run 'hyperdev generate --all' to update."
    exit 1
fi

echo "✅ All HyperDev validations passed."
exit 0
```

#### Pre-push Hooks

```bash
#!/bin/sh
# .git/hooks/pre-push

echo "Running HyperDev pre-push quality gates..."

# Run full quality gate suite
if ! hyperdev quality-gate --all --strict; then
    echo "❌ Quality gates failed. Push blocked."
    exit 1
fi

# Verify template compatibility
if ! hyperdev compatibility-check --remote; then
    echo "❌ Template compatibility issues detected."
    exit 1
fi

echo "✅ All quality gates passed. Push allowed."
exit 0
```

## Team Collaboration Integration

### Code Review Integration

```yaml
# .github/workflows/pr-review.yml
name: PR Review with HyperDev

on:
  pull_request:
    types: [opened, synchronize]

jobs:
  hyperdev-review:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 0
          
      - name: HyperDev Template Review
        run: |
          # Get changed templates
          CHANGED_TEMPLATES=$(git diff --name-only HEAD~1 HEAD | grep -E '\.(template\.yml|ejs\.t)$' || true)
          
          if [ ! -z "$CHANGED_TEMPLATES" ]; then
            echo "Reviewing changed templates..."
            hyperdev review-templates --changed-only --format json > template-review.json
            
            # Generate review comments
            hyperdev generate-review-comments template-review.json > review-comments.md
          fi
          
      - name: Post Review Comments
        if: success()
        uses: actions/github-script@v7
        with:
          script: |
            const fs = require('fs');
            
            if (fs.existsSync('review-comments.md')) {
              const comments = fs.readFileSync('review-comments.md', 'utf8');
              
              github.rest.issues.createComment({
                issue_number: context.issue.number,
                owner: context.repo.owner,
                repo: context.repo.repo,
                body: comments
              });
            }
```

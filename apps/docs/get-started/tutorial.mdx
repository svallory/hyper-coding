## Part 4: Custom Template Creation

Now comes the exciting part - creating your own custom template! We'll build a template that matches your specific project patterns.

### Step 4.1: Analyze Your Project Patterns

First, let's identify patterns in your existing code (or create some standards):

```bash
# Create a sample component to use as a pattern
mkdir -p patterns/
cat > patterns/SampleComponent.tsx << 'EOF'
import React, { useState } from 'react';
import { clsx } from 'clsx';
import styles from './SampleComponent.module.css';

export interface SampleComponentProps {
  className?: string;
  variant?: 'primary' | 'secondary';
  children: React.ReactNode;
  onClick?: () => void;
}

/**
 * Sample component following our project patterns
 */
export const SampleComponent: React.FC<SampleComponentProps> = ({
  className,
  variant = 'primary',
  children,
  onClick
}) => {
  const [isActive, setIsActive] = useState(false);

  return (
    <div 
      className={clsx(
        styles.container,
        styles[variant],
        { [styles.active]: isActive },
        className
      )}
      onClick={onClick}
    >
      {children}
    </div>
  );
};
EOF
```

## Part 5: Advanced Features

Let's explore HyperDev's advanced features: actions, codemods, and recipe composition.

### Step 5.1: Understanding Actions

Actions are TypeScript functions that can perform complex operations beyond template rendering:

```bash
# Create an action that updates your main index file
mkdir -p cookbooks/my-components/actions

cat > cookbooks/my-components/actions/update-index.ts << 'EOF'
import { readFileSync, writeFileSync } from 'fs';
import path from 'path';

export interface UpdateIndexArgs {
  componentName: string;
  componentPath: string;
}

/**
 * Updates the main components index file to export the new component
 */
export async function updateIndex(args: UpdateIndexArgs): Promise<void> {
  const indexPath = path.join('src', 'components', 'index.ts');
  
  // Read existing index file or create new content
  let content = '';
  try {
    content = readFileSync(indexPath, 'utf-8');
  } catch (error) {
    // File doesn't exist, start with empty content
    content = '// Auto-generated component exports\n\n';
  }
  
  // Add export for new component
  const exportLine = `export { ${args.componentName} } from './${args.componentName}';`;
  
  if (!content.includes(exportLine)) {
    content += exportLine + '\n';
    writeFileSync(indexPath, content);
    console.log(`âœ… Added ${args.componentName} to component index`);
  } else {
    console.log(`â„¹ï¸  ${args.componentName} already exported in index`);
  }
}
EOF
```

### Step 5.4: Introduction to CodeMods

CodeMods modify existing code. Let's create one that adds your new component to a storybook file:

```bash
# Create a codemod that adds Storybook stories
mkdir -p cookbooks/my-components/codemods

cat > cookbooks/my-components/codemods/add-storybook.ts << 'EOF'
import { readFileSync, writeFileSync } from 'fs';

export interface AddStorybookArgs {
  componentName: string;
}

/**
 * Adds a basic Storybook story for the component
 */
export async function addStorybook(args: AddStorybookArgs): Promise<void> {
  const storyPath = `src/components/${args.componentName}/${args.componentName}.stories.tsx`;
  
  const storyContent = `import type { Meta, StoryObj } from '@storybook/react';
import { ${args.componentName} } from './${args.componentName}';

const meta: Meta<typeof ${args.componentName}> = {
  title: 'Components/${args.componentName}',
  component: ${args.componentName},
  parameters: {
    layout: 'centered',
  },
  tags: ['autodocs'],
};

export default meta;
type Story = StoryObj<typeof meta>;

export const Default: Story = {
  args: {
    children: 'Hello World',
  },
};

export const Primary: Story = {
  args: {
    variant: 'primary',
    children: 'Primary variant',
  },
};

export const Secondary: Story = {
  args: {
    variant: 'secondary',
    children: 'Secondary variant',
  },
};
`;

  writeFileSync(storyPath, storyContent);
  console.log(`âœ… Created Storybook story for ${args.componentName}`);
}
EOF
```

### Step 5.5: Recipe Composition

You can create recipes that use other recipes. Let's create a "feature" recipe that generates multiple related components:

```bash
# Create a feature recipe
cat > cookbooks/feature-generator/recipe.yml << 'EOF'
name: feature
description: Generate a complete feature with multiple components

variables:
  - name: feature_name
    type: string
    description: Feature name (kebab-case)
    required: true
  
  - name: components
    type: array
    description: List of components to generate
    default: ["List", "Item", "Form"]

steps:
  - tool: recipe
    name: generate-components
    config:
      recipe: ../my-components/recipe.yml
      forEach: "{{ components }}"
      args:
        name: "{{ feature_name | capitalize }}{{ item }}"
        variant_support: true
        interactive: true
        description: "{{ item }} component for {{ feature_name }} feature"
  
  - tool: template
    name: create-feature-index
    config:
      source: feature-index.ts.liquid
      target: "src/features/{{ feature_name }}/index.ts"
EOF

# Create the feature index template
cat > cookbooks/feature-generator/feature-index.ts.liquid << 'EOF'
// {{ feature_name | capitalize }} Feature Components
{% for component in components %}
export { {{ feature_name | capitalize }}{{ component }} } from '../../components/{{ feature_name | capitalize }}{{ component }}';
{% endfor %}
EOF
```

```bash

  ],
  
  // Template engine preferences
  engines: {
    default: 'liquidjs',   // Primary engine
```

## Template Best Practices

# Next.js Integration

Configure Next.js to work seamlessly with HyperDev for automatic generation of API routes, components, and pages.

## Next.js Plugin Configuration

Set up the HyperDev Next.js plugin:

```javascript
// next.config.js
const { withHyperDev } = require('@hyperdev/next-plugin')

/** @type {import('next').NextConfig} */
const nextConfig = {
  // ... other config
  experimental: {
    serverComponentsExternalPackages: ['@hyperdev/core']
  }
}

module.exports = withHyperDev({
  // HyperDev configuration
  hyperdev: {
    // API route generation
    api: {
      autoGenerate: true,
      pattern: 'pages/api/**/*.template.yml',
      typescript: true
    },
    // Component generation
    components: {
      autoGenerate: true,
      pattern: 'components/**/*.template.yml',
      defaultProps: true
    },
    // Page generation
    pages: {
      autoGenerate: true,
      pattern: 'pages/**/*.template.yml',
      seo: true
    },
    // Quality gates
    qualityGates: {
      enabled: true,
      onBuild: true,
      rules: ['next-seo', 'accessibility', 'performance']
    }
  }
})(nextConfig)
```

## API Routes Integration

Generate Next.js API routes with HyperDev:

```yaml
# templates/api-route.template.yml
name: "Next.js API Route"
description: "Generate a Next.js API route with validation"
variables:
  - name: routeName
    description: "API route name"
    type: string
  - name: method
    description: "HTTP method"
    type: enum
    options: ["GET", "POST", "PUT", "DELETE"]
```

```typescript
// templates/api-route.ejs.t
---
to: pages/api/<%= routeName %>.ts
---
import type { NextApiRequest, NextApiResponse } from 'next'

type Data = {
  message: string
}

export default function handler(
  req: NextApiRequest,
  res: NextApiResponse<Data>
) {
  if (req.method === '<%= method %>') {
    res.status(200).json({ message: 'Success' })
  } else {
    res.setHeader('Allow', ['<%= method %>'])
    res.status(405).end(`Method ${req.method} Not Allowed`)
  }
}
```

## Component Generation

Generate Next.js components with TypeScript:

```yaml
# templates/component.template.yml
name: "Next.js Component"
description: "Generate a Next.js React component"
variables:
  - name: componentName
    description: "Component name"
    type: string
  - name: withProps
    description: "Include props interface"
    type: boolean
    default: true
```

```tsx
// templates/component.ejs.t
---
to: components/<%= componentName %>.tsx
---
<% if (withProps) { %>
interface <%= componentName %>Props {
  className?: string
}
<% } %>

export default function <%= componentName %>(<% if (withProps) { %>props: <%= componentName %>Props<% } %>) {
  return (
    <div<% if (withProps) { %> className={props.className}<% } %>>
      <h1><%= componentName %></h1>
    </div>
  )
}
```

## Page Generation

Generate Next.js pages with SEO:

```tsx
// templates/page.ejs.t
---
to: pages/<%= pageName %>.tsx
---
import Head from 'next/head'
import type { NextPage } from 'next'

const <%= h.changeCase.pascalCase(pageName) %>Page: NextPage = () => {
  return (
    <>
      <Head>
        <title><%= h.changeCase.titleCase(pageName) %></title>
        <meta name="description" content="<%= description || 'Generated page' %>" />
      </Head>
      
      <main>
        <h1><%= h.changeCase.titleCase(pageName) %></h1>
        <p>This page was generated by HyperDev.</p>
      </main>
    </>
  )
}

export default <%= h.changeCase.pascalCase(pageName) %>Page
```

## Development Scripts

Package.json scripts for Next.js and HyperDev:

```json
{
  "scripts": {
    "dev": "concurrently \"hyperdev dev --watch\" \"next dev\"",
    "build": "hyperdev validate --all && next build",
    "start": "next start",
    "gen:api": "hyperdev gen api-route",
    "gen:component": "hyperdev gen component",
    "gen:page": "hyperdev gen page"
  }
}
```

## TypeScript Configuration

Configure TypeScript for HyperDev and Next.js:

```json
// tsconfig.json
{
  "compilerOptions": {
    // ... other options
    "paths": {
      "@/components/*": ["./components/*"],
      "@/pages/*": ["./pages/*"],
      "@/generated/*": ["./generated/*"]
    }
  },
  "include": [
    "next-env.d.ts",
    "**/*.ts",
    "**/*.tsx",
    ".hyperdev/generated/**/*"
  ]
}
```

This integration provides seamless Next.js development with automatic API route, component, and page generation through HyperDev templates.

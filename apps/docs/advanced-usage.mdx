# Advanced Usage Patterns

Unlock HyperDev's full potential with advanced template composition, enterprise automation, and sophisticated workflow patterns designed for power users and large-scale deployments.

## Advanced Template Composition

### Multi-Template Orchestration

Execute complex workflows that coordinate multiple templates with dependencies and conditional logic.

```yaml
# hyperdev.recipe.yml
name: "Full-Stack Component"
description: "Generate complete full-stack feature with frontend, backend, and tests"

steps:
  - name: "backend-api"
    template: "express-api"
    config:
      endpoint: "{{ feature.name }}"
      model: "{{ feature.model }}"
    
  - name: "frontend-component" 
    template: "react-component"
    depends_on: ["backend-api"]
    config:
      name: "{{ feature.name | capitalize }}"
      api_endpoint: "{{ steps.backend-api.output.endpoint }}"
      
  - name: "integration-tests"
    template: "playwright-tests"
    depends_on: ["backend-api", "frontend-component"]
    config:
      test_target: "{{ feature.name }}"
```

### Template Inheritance Hierarchies

Build sophisticated template inheritance systems for consistent patterns across your organization.

```yaml
# templates/base/component.yml
name: "Base Component"
variables:

    type: string
    required: true
  styling:

    options: ["styled-components", "emotion", "css-modules"]
    default: "styled-components"

hooks:
  pre_generate:
    - action: "validate_component_name"
    - action: "check_dependencies"
```

```yaml

    default: "vitest"

```

## Custom Extensions

### Plugin Development

Create sophisticated plugins for organization-specific workflows.

```typescript
// plugins/enterprise-security/index.ts
import { HyperdevPlugin, PluginContext, GenerationContext } from '@hyperdev/hyperdev';

export class EnterpriseSecurityPlugin implements HypergenPlugin {
  name = 'enterprise-security';
  version = '1.2.0';

  async initialize(context: PluginContext) {
    context.registerHook('pre_generation', this.validateSecurity.bind(this));
    context.registerHook('post_generation', this.scanGenerated.bind(this));
    context.registerAction('security_scan', this.performSecurityScan.bind(this));
    context.registerValidator('pii_check', this.validatePII.bind(this));
  }

  async validateSecurity(context: GenerationContext) {
    // Check template source trustworthiness
    const trustLevel = await this.assessTemplateTrust(context.template);
    if (trustLevel < context.config.security.min_trust_level) {
      throw new SecurityValidationError(
        `Template trust level ${trustLevel} below required ${context.config.security.min_trust_level}`
      );
    }

    // Validate generation parameters for sensitive data
    await this.scanParametersForSensitiveData(context.parameters);
  }

  async performSecurityScan(files: GeneratedFile[]) {
    const results = await Promise.all([
      this.scanForSecrets(files),
      this.scanForVulnerabilities(files), 
      this.checkCompliance(files),
      this.validateDataClassification(files)
    ]);

    return this.consolidateSecurityResults(results);
  }
}
```

### Advanced Template Testing

Build comprehensive testing frameworks for template validation.

```typescript
// testing/template-test-framework.ts
export class TemplateTestSuite {
  constructor(private engine: HypergenEngine) {}

  async runComprehensiveTests(templatePath: string) {
    const results = await Promise.all([
      this.testSyntaxValidation(templatePath),
      this.testVariableValidation(templatePath),
      this.testOutputValidation(templatePath),
      this.testPerformanceBenchmarks(templatePath),
      this.testSecurityCompliance(templatePath),
      this.testCrossEnvironmentCompatibility(templatePath)
    ]);

    return this.generateTestReport(results);
  }

  async testOutputValidation(templatePath: string) {
    const testCases = await this.loadTestCases(templatePath);
    
    for (const testCase of testCases) {
      const result = await this.engine.generateFromTemplate(
        templatePath, 
        testCase.parameters
      );
      
      // Validate structure
      await this.validateFileStructure(result.files, testCase.expected.structure);
      
      // Validate content
      await this.validateContent(result.files, testCase.expected.content);
      
      // Validate compilation
      if (testCase.compile_test) {
        await this.validateCompilation(result.files, testCase.compile_test);
      }
    }
  }
}
```

### Multi-Tenant Repository Management

Manage templates across multiple teams and organizations.

```typescript
// enterprise/multi-tenant-manager.ts
export class MultiTenantTemplateManager {
  constructor(private config: EnterpriseConfig) {}

  async resolveTemplate(templateName: string, context: RequestContext) {
    const tenant = await this.identifyTenant(context);
    const permissions = await this.getPermissions(tenant, templateName);
    
    // Check access rights
    this.validateAccess(permissions, context.operation);
    
    // Resolve template with tenant-specific overrides
    const template = await this.loadTemplate(templateName, {
      tenant_overrides: tenant.template_overrides,
      compliance_rules: tenant.compliance_requirements,
      custom_variables: tenant.default_variables
    });

    return this.applyTenantPolicies(template, tenant);
  }

  async trackUsage(tenant: string, template: string, context: GenerationContext) {
    await this.analytics.record({
      event: 'template_generation',
      tenant,
      template,
      user: context.user.id,
      timestamp: new Date(),
      parameters: this.sanitizeForAudit(context.parameters),
      output_files: context.result.files.length,
      generation_time: context.performance.total_time,
      compliance_tags: context.template.compliance_tags
    });
  }
}
```

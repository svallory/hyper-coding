## Hygen to HyperDev Migration

### Automated Migration Tool

Use the built-in migration tool for seamless Hygen to HyperDev conversion.

```bash
# Install HyperDev globally
npm install -g @hyperdev/hypergen

# Run automated migration
hypergen migrate hygen \
  --source="./_templates" \
  --output="./templates" \
  --config-output="./hypergen.config.js" \
  --preserve-structure \
  --upgrade-actions

# Validate migrated templates
hypergen validate ./templates --strict

# Test migrated templates
hypergen test ./templates --all
```

### Template Structure Migration

Hygen templates migrate directly with enhanced features.

<CodeGroup>
```yaml Before (Hygen)
# _templates/component/new/index.js
module.exports = {
  prompt: ({ inquirer }) => {
    const questions = [
      {
        type: 'input',
        name: 'name',
        message: 'Component name?'
      }
    ];
    return inquirer.prompt(questions);
  }
};
```

```yaml After (HyperDev)
# templates/component/template.yml
name: "React Component"
version: "1.0.0"

    type: string

    to: "src/components/{{ name }}/{{ name }}.tsx"
```

### Action System Upgrade

Hygen actions are enhanced in HyperDev's V8 action system.

<CodeGroup>
```javascript Before (Hygen Action)
// _templates/component/new/component.ejs.t
---
to: src/components/<%= name %>/<%= name %>.tsx
---
import React from 'react';

const <%= name %>: React.FC = () => {
  return <div><%= name %> Component</div>;
};

export default <%= name %>;
```

```typescript After (HyperDev Action)
// templates/component/actions/component.ts
import { Action, ActionContext } from '@hyperdev/hypergen';

@Action('generate-component')
export class GenerateComponentAction {
  async execute(context: ActionContext) {
    const { name, typescript } = context.parameters;
    
    // Enhanced validation
    if (!name.match(/^[A-Z][A-Za-z0-9]*$/)) {
      throw new Error('Component name must be PascalCase');
    }
    
    // Generate files with enhanced logic
    await context.generateFile('component.tsx.ejs', {
      to: `src/components/${name}/${name}.tsx`,
      parameters: { name, typescript }
    });
    
    // Generate tests automatically
    if (context.config.include_tests) {
      await context.generateFile('component.test.tsx.ejs', {
        to: `src/components/${name}/${name}.test.tsx`,
        parameters: { name }
      });
    }
    
    // Update index file
    await context.injectIntoFile('src/components/index.ts', {
      pattern: /export \* from/,
      template: `export * from './${name}/${name}';`
    });
  }
}
```
</CodeGroup>

### Configuration Migration

Hygen configuration migrates to HyperDev's enhanced config system.

```javascript
// hypergen.config.js (migrated from .hygen.js)
export default {
  // Migrated template directories
  template_dirs: ['./templates'],
  
  // Enhanced configuration options
  generator: {
    // Hygen compatibility mode
    hygen_compatibility: true,
    
    // Enhanced features
    typescript_actions: true,
    recipe_system: true,
    multi_template_support: true
  },
  
  // Migration-specific settings
  migration: {
    source: 'hygen',
    preserve_file_structure: true,
    upgrade_actions: true,
    
    // Compatibility mappings
    action_mappings: {
      'add': 'generate_file',
      'inject': 'inject_into_file',
      'append': 'append_to_file'
    }
  },
  
  // Enhanced validation
  validation: {
    strict_mode: true,
    require_typescript: true,
    security_scan: true
  }
};
```

<CodeGroup>
```javascript Before (Yeoman)

    this.answers = await this.prompt([

        type: 'input',

        type: 'list',
```

## Plop Migration

### Plopfile Configuration Conversion

Convert Plop configurations to HyperDev's enhanced system.

<CodeGroup>
```javascript Before (Plop)
// plopfile.js
module.exports = function (plop) {
  plop.setGenerator('component', {
    description: 'React component generator',
    prompts: [{
      type: 'input',
      name: 'name',
      message: 'Component name?'
    }],
    actions: [{
      type: 'add',
      path: 'src/components/{{pascalCase name}}/{{pascalCase name}}.tsx',
      templateFile: 'plop-templates/component.hbs'
    }]
  });
};
```

```yaml After (HyperDev)
# templates/component/template.yml
name: "React Component"
version: "1.0.0"
description: "React component generator (migrated from Plop)"

variables:
  name:
    type: string
    required: true
    prompt: "Component name?"
    validation: /^[a-zA-Z][a-zA-Z0-9]*$/

# Enhanced template processing
computed_values:
  pascal_name: "{{ name | pascalCase }}"
  kebab_name: "{{ name | kebabCase }}"

files:
  - from: "component.tsx.ejs"
    to: "src/components/{{ computed_values.pascal_name }}/{{ computed_values.pascal_name }}.tsx"
    
  # Additional files not in original Plop generator
  - from: "component.test.tsx.ejs"
    to: "src/components/{{ computed_values.pascal_name }}/{{ computed_values.pascal_name }}.test.tsx"
    condition: "include_tests !== false"
    
  - from: "index.ts.ejs"
    to: "src/components/{{ computed_values.pascal_name }}/index.ts"

# Enhanced actions
actions:
  post_generate:
    - action: "update_component_index"
      parameters:
        component_name: "{{ computed_values.pascal_name }}"
```
</CodeGroup>

### Handlebars to EJS Migration

Automatic template conversion from Handlebars to EJS with enhanced features.

```bash
# Convert Handlebars templates to EJS
hypergen migrate templates \
  --source="./plop-templates" \
  --from="handlebars" \
  --to="ejs" \
  --output="./templates" \
  --preserve-logic \
  --upgrade-helpers

# Validate converted templates
hypergen validate ./templates \
  --template-engine="ejs" \
  --check-helpers
```

## Custom Script Migration

### Pattern Recognition and Template Generation

Analyze existing custom scripts and generate equivalent templates.

```bash
# Analyze custom generation scripts
hypergen analyze scripts \
  --source="./scripts/generate" \
  --recursive \
  --output-report="./script-analysis.json" \
  --detect-patterns

# Generate templates from patterns
hypergen migrate scripts \
  --source="./scripts/generate" \
  --output="./templates" \
  --create-recipes \
  --preserve-logic \
  --upgrade-features
```

### Script Pattern Analysis

HyperDev analyzes common patterns in custom scripts.

<CodeGroup>
```bash Before (Custom Script)
#!/bin/bash
# scripts/create-component.sh

COMPONENT_NAME=$1
COMPONENT_DIR="src/components/$COMPONENT_NAME"

mkdir -p "$COMPONENT_DIR"

cat > "$COMPONENT_DIR/$COMPONENT_NAME.tsx" << EOF
import React from 'react';

interface ${COMPONENT_NAME}Props {
  // Props go here
}

const $COMPONENT_NAME: React.FC<${COMPONENT_NAME}Props> = () => {
  return <div>$COMPONENT_NAME Component</div>;
};

export default $COMPONENT_NAME;
EOF

echo "Component $COMPONENT_NAME created successfully!"
```

```yaml After (HyperDev Template)
# templates/component/template.yml (generated from script analysis)
name: "React Component"
version: "1.0.0" 
description: "Generate React component (migrated from create-component.sh)"

variables:
  name:
    type: string
    required: true
    prompt: "Component name?"
    validation: /^[A-Z][A-Za-z0-9]*$/

computed_values:
  component_dir: "src/components/{{ name }}"
  props_interface: "{{ name }}Props"

files:
  - from: "component.tsx.ejs"
    to: "{{ computed_values.component_dir }}/{{ name }}.tsx"
    create_dirs: true

actions:
  post_generate:
    - action: "log_success"
      message: "Component {{ name }} created successfully!"
```
</CodeGroup>
